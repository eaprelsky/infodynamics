def calculate_information_entropy_loss(original_content: str, transformed_content: str) -> float:
    """
    –†–∞—Å—á–µ—Ç –ø–æ—Ç–µ—Ä—å –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ —á–µ—Ä–µ–∑ –∏–∑–º–µ–Ω–µ–Ω–∏–µ —ç–Ω—Ç—Ä–æ–ø–∏–∏
    """
    # –ò—Å—Ö–æ–¥–Ω–∞—è —ç–Ω—Ç—Ä–æ–ø–∏—è

    # –≠–Ω—Ç—Ä–æ–ø–∏—è –ø–æ—Å–ª–µ —Ç—Ä–∞–Ω—Å—Ñ–æ—Ä–º–∞—Ü–∏–∏

    # –ü–æ—Ç–µ—Ä–∏ –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ (–≤—Å–µ–≥–¥–∞ –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–µ –∏–ª–∏ –Ω—É–ª–µ–≤—ã–µ)
    entropy_loss = max(0, H_original - H_transformed)
    
    # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –∫ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–º –ø–æ—Ç–µ—Ä—è–º

    return relative_loss

def calculate_semantic_drift(original_embedding: np.ndarray, transformed_embedding: np.ndarray) -> float:
    """
    –°–µ–º–∞–Ω—Ç–∏—á–µ—Å–∫–∏–π –¥—Ä–∏—Ñ—Ç —á–µ—Ä–µ–∑ –∫–æ—Å–∏–Ω—É—Å–Ω–æ–µ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ
    """
    cosine_sim = np.dot(original_embedding, transformed_embedding) / (
        np.linalg.norm(original_embedding) * np.linalg.norm(transformed_embedding)
    )
    
    semantic_drift = 1 - cosine_sim  # [0, 2], –≥–¥–µ 0 - –Ω–µ—Ç –¥—Ä–∏—Ñ—Ç–∞, 2 - –ø–æ–ª–Ω–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ
    
    return semantic_drift

def transformation_quality_score(
    entropy_loss: float, 
    semantic_drift: float, 
    factual_preservation: float,
    audience_fit: float

    """

    w_semantic = 0.30     # –°–µ–º–∞–Ω—Ç–∏—á–µ—Å–∫–∞—è –±–ª–∏–∑–æ—Å—Ç—å
    w_factual = 0.25      # –§–∞–∫—Ç–∏—á–µ—Å–∫–∞—è —Ç–æ—á–Ω–æ—Å—Ç—å
    w_audience = 0.20     # –°–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –∞—É–¥–∏—Ç–æ—Ä–∏–∏

    entropy_score = 1 - entropy_loss
    semantic_score = 1 - min(semantic_drift / 2, 1)  # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è
    
    total_score = (
        w_entropy * entropy_score +
