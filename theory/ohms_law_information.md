    E_available: float,         # –î–æ—Å—Ç—É–ø–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)
    neural_efficiency: float = 0.25,  # –ù–µ–π—Ä–æ–Ω–Ω–∞—è —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç—å
    frequency: float = 0        # –ß–∞—Å—Ç–æ—Ç–∞ (–¥–ª—è AC –∞–Ω–∞–ª–∏–∑–∞)

        P_ideal = U_info * I_ideal

        I_actual = I_ideal
        P_actual = P_ideal

        performance_factor = E_available / E_neural_required

    E_stored_C = 0.5 * C_info * U_info ** 2    # –≠–Ω–µ—Ä–≥–∏—è –µ–º–∫–æ—Å—Ç–∏
    E_total_stored = E_stored_L + E_stored_C

        'power_ideal': P_ideal,

        'neural_efficiency': neural_efficiency,
        'system_efficiency': E_useful / E_consumed if E_consumed > 0 else 0,
        'energy_balance_error': energy_balance_check
    }
